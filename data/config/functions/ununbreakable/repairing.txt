# Beispiele 端ber Reparieren
# Um ein Material hinzuzuf端gen, muss man einfach einen Command hinzuf端gen, der dem custom Item den repairing Tag zuweist, wenn das Reparier-Material verf端gbar ist,
# einen Command um die repairUses festzulegen und einen Command der dieses Material als "benutzt" markiert! (siehe Beispiele)
# Beispiele:

execute as @s[nbt={Item:{tag:{repairItems:1s}}}] if entity @e[type=item,nbt={Item:{id:"minecraft:glass"}},distance=..0.5] run tag @s add repairing
execute as @s[nbt={Item:{tag:{repairItems:1s}}}] if entity @e[type=item,nbt={Item:{id:"minecraft:purple_stained_glass"}},distance=..0.5] run tag @s add repairing

execute as @s[nbt={Item:{tag:{repairItems:2s}}}] if entity @e[type=item,nbt={Item:{id:"minecraft:diamond",tag:{display:{Name:"{\"text\":\"magischer Reparaturstein\"}"}}}},distance=..0.5] run tag @s add repairing 

#mark

execute as @s[nbt={Item:{tag:{repairItems:1s}}}] unless entity @e[tag=offer,distance=..0.5] run tag @e[type=item,nbt={Item:{id:"minecraft:glass"}},distance=..0.5] add offer
execute as @s[nbt={Item:{tag:{repairItems:1s}}}] unless entity @e[tag=offer,distance=..0.5] run tag @e[type=item,nbt={Item:{id:"minecraft:purple_stained_glass"}},distance=..0.5] add offer

execute as @s[nbt={Item:{tag:{repairItems:2s}}}] unless entity @e[tag=offer,distance=..0.5] run tag @e[type=item,nbt={Item:{id:"minecraft:diamond",tag:{display:{Name:"{\"text\":\"magischer Reparaturstein\"}"}}}},distance=..0.5] add offer
